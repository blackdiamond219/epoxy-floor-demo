  <h1>Garage Epoxy Mockup Tool</h1>

  <p>Upload your garage photo, enter your preferred colors and pattern, and see your epoxy mockup.</p>


  <input type="file" id="imageUpload" accept="image/*">

  <input type="text" id="preferences" placeholder="Enter colors and pattern (e.g., red black galaxy)">

  <button onclick="uploadImage()">Generate Mockup</button>


  <div id="result"></div>


  <script>

    const cloudName = "dkn8ynrud";

    const uploadPreset = "YOUR_UPLOAD_PRESET"; // Replace with your actual upload preset

    const webhookUrl = "https://mcooper21.app.n8n.cloud/webhook/garage-mockup";


    async function uploadImage() {

      const fileInput = document.getElementById("imageUpload");

      const preferences = document.getElementById("preferences").value;


      if (!fileInput.files[0]) {

        alert("Please select an image.");

        return;

      }


      const formData = new FormData();

      formData.append("file", fileInput.files[0]);

      formData.append("upload_preset", uploadPreset);


      // Upload to Cloudinary

      const cloudinaryResponse = await fetch(`https://api.cloudinary.com/v1_1/${cloudName}/image/upload`, {

        method: "POST",

        body: formData

      });


      const cloudinaryData = await cloudinaryResponse.json();

      const imageUrl = cloudinaryData.secure_url;


      // Send to n8n webhook

      const response = await fetch(webhookUrl, {

        method: "POST",

        headers: { "Content-Type": "application/json" },

        body: JSON.stringify({

          image_url: imageUrl,

          preferences: preferences

        })

      });


      const result = await response.json();

      document.getElementById("result").innerHTML = `<h2>Mockup Result:</h2><img src="${result.image_url}" alt="Mockup">`;

    }

  </script>
